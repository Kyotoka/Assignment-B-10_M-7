Odd or Even
// ODD OR EVEN ?
function isEven(num) {
    let conclusion;
    if (num % 2 === 0) {
        conclusion = "Even";
    } else {
        conclusion = "Odd";
    }
    return conclusion;
}

console.log("Conclusion = Number Is" + isEven(7));
console.log("Conclusion = Number Is" + isEven(128));
Results:
Conclusion = Number Is Odd
Conclusion = Number Is Even
Celsius to Fahrenheit
// CELCIUS TO FARENHEIGHT
function CtoF(C) {
    let F;
    F = ((C * (9 / 5)) + 32);
    return F;
}

console.log("TEMP = " + CtoF(34) + "°F");
Temperature:
TEMP = 93.2°F
