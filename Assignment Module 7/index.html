Odd or Even Checker

// ODD OR EVEN ?
function isEven(num) {
    let conclusion;
    if (num % 2 === 0) {
        conclusion = "Even";
    } else {
        conclusion = "Odd";
    }
    return conclusion;
}

console.log("Conclusion = Number Is :" + isEven(7));
console.log("Conclusion = Number Is :" + isEven(128));
Output:
Conclusion = Number Is :Odd
Conclusion = Number Is :Even

Celsius to Fahrenheit

// CELCIUS TO FARENHEIGHT
function CtoF(C) {
    let F;
    F = ((C * (9 / 5)) + 32);
    return F;
}

console.log("TEMP = " + CtoF(34) + "°F");
Output: TEMP = 93.2°F

Multiply with Type Check

// MULTIPLY NUMBER
function mlt(x, y) {
    if (typeof x !== "number")
        return "1st entry is not a number"
    else if (typeof y !== "number")
        return "2nd entry is not a number"
    else return x * y;
}
console.log(mlt(10, "s"));
console.log(mlt(12, 7));
console.log(mlt("y", 7));
Output:
2nd entry is not a number
84
1st entry is not a number

Minutes to Seconds

// MIN TO SEC
function mTs(m) {
    let s;
    s = m * 60;
    return s;
} 
console.log(mTs(5) + "s");
Output: 300s

Check String Contains

// CHECK IF THE SENTENCE HAS A CERTAIN WORD
function checkStr(sentence, word) {
    return sentence.includes(word);
}
console.log(checkStr("Hi there", "boy"));
console.log(checkStr("Hi there boy", "boy"));
Output:
false
true
